function getNotifIcon(type: string, size = 20) {

  if (type.includes('MEETING')) return <Calendar size={size} />;
  if (type.includes('DOCUMENT') || type === 'CONTACT_RECEIVED') return <FileText size={size} />;
  return <Bell size={size} />;
}
